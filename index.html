C++ Programs for Learning

1) Write a program to find the largest of three numbers using nested if else.

#include <iostream> using namespace std; int main() { int a, b, c; cout << "Enter three numbers: "; cin >> a >> b >> c; if (a > b) { if (a > c) cout << "Largest: " << a; else cout << "Largest: " << c; } else { if (b > c) cout << "Largest: " << b; else cout << "Largest: " << c; } return 0; }

2) Write a program to implement Arithmetic calculator using switch.

#include <iostream> using namespace std; int main() { char op; float num1, num2; cout << "Enter operator: +, -, *, /: "; cin >> op; cout << "Enter two operands: "; cin >> num1 >> num2; switch(op) { case '+': cout << num1 << " + " << num2 << " = " << num1 + num2; break; case '-': cout << num1 << " - " << num2 << " = " << num1 - num2; break; case '*': cout << num1 << " * " << num2 << " = " << num1 * num2; break; case '/': if(num2 != 0) cout << num1 << " / " << num2 << " = " << num1 / num2; else cout << "Division by zero!"; break; default: cout << "Invalid operator"; break; } return 0; }

3) Write a program to calculate the factorial of a number using a function in C++.

#include <iostream> using namespace std; int factorial(int n) { if (n == 0) return 1; else return n * factorial(n - 1); } int main() { int num; cout << "Enter a number: "; cin >> num; cout << "Factorial of " << num << " is " << factorial(num); return 0; }

4) Write a program to implement function overloading in C++.

#include <iostream> using namespace std; int add(int a, int b) { return a + b; } double add(double a, double b) { return a + b; } int main() { int x = 10, y = 20; double m = 2.5, n = 3.5; cout << "Sum of integers: " << add(x, y) << endl; cout << "Sum of doubles: " << add(m, n) << endl; return 0; }

5) Write a program to implement a class in C++.

#include <iostream> using namespace std; class Person { public: string name; int age; void display() { cout << "Name: " << name << ", Age: " << age << endl; } }; int main() { Person p1; p1.name = "John"; p1.age = 30; p1.display(); return 0; }

6) Write a program to implement scope resolution operator in C++.

#include <iostream> using namespace std; int x = 10; int main() { int x = 20; cout << "Local x: " << x << endl; cout << "Global x: " << ::x << endl; return 0; }

7) Write a program to implement a class with arrays as data members.

#include <iostream> using namespace std; class Array { private: int arr[5]; public: void insert(int index, int value) { if(index >= 0 && index < 5) arr[index] = value; } void display() { for(int i = 0; i < 5; i++) cout << arr[i] << " "; cout << endl; } }; int main() { Array a; a.insert(0, 10); a.insert(1, 20); a.insert(2, 30); a.insert(3, 40); a.insert(4, 50); a.display(); return 0; }

8) Write a program to implement the use of static data member.

#include <iostream> using namespace std; class Counter { public: static int count; Counter() { count++; } }; int Counter::count = 0; int main() { Counter c1, c2, c3; cout << "Count: " << Counter::count << endl; return 0; }

9) Write a program to illustrate the use of object arrays.

#include <iostream> using namespace std; class Student { public: string name; int age; void getData() { cout << "Enter name and age: "; cin >> name >> age; } void display() { cout << "Name: " << name << ", Age: " << age << endl; } }; int main() { Student students[3]; for(int i = 0; i < 3; i++) { students[i].getData(); } for(int i = 0; i < 3; i++) { students[i].display(); } return 0; }
