C++ Programs for Learning

1) Write a program to find the largest of three numbers using nested if else.

#include <iostream> using namespace std; int main() { int a, b, c; cout << "Enter three numbers: "; cin >> a >> b >> c; if (a > b) { if (a > c) cout << "Largest: " << a; else cout << "Largest: " << c; } else { if (b > c) cout << "Largest: " << b; else cout << "Largest: " << c; } return 0; }

2) Write a program to implement Arithmetic calculator using switch.

#include <iostream> using namespace std; int main() { char op; float num1, num2; cout << "Enter operator: +, -, *, /: "; cin >> op; cout << "Enter two operands: "; cin >> num1 >> num2; switch(op) { case '+': cout << num1 << " + " << num2 << " = " << num1 + num2; break; case '-': cout << num1 << " - " << num2 << " = " << num1 - num2; break; case '*': cout << num1 << " * " << num2 << " = " << num1 * num2; break; case '/': if(num2 != 0) cout << num1 << " / " << num2 << " = " << num1 / num2; else cout << "Division by zero!"; break; default: cout << "Invalid operator"; break; } return 0; }

3) Write a program to calculate the factorial of a number using a function in C++.

#include <iostream> using namespace std; int factorial(int n) { if (n == 0) return 1; else return n * factorial(n - 1); } int main() { int num; cout << "Enter a number: "; cin >> num; cout << "Factorial of " << num << " is " << factorial(num); return 0; }

4) Write a program to implement function overloading in C++.

#include <iostream> using namespace std; int add(int a, int b) { return a + b; } double add(double a, double b) { return a + b; } int main() { int x = 10, y = 20; double m = 2.5, n = 3.5; cout << "Sum of integers: " << add(x, y) << endl; cout << "Sum of doubles: " << add(m, n) << endl; return 0; }

6) Write a program to implement scope resolution operator in C++.

7) Write a program to implement a class with arrays as data members.

10) Write a program to illustrate the use of the friend function

20) Write a program to implement single inheritance

#include <iostream> using namespace std; class Base { public: void display() { cout << "Base class display function" << endl; } }; class Derived : public Base { public: void show() { cout << "Derived class show function" << endl; } }; int main() { Derived obj; obj.display(); obj.show(); return 0; }

21) Write a program to implement multilevel inheritance

#include <iostream> using namespace std; class Base { public: void display() { cout << "Base class display function" << endl; } }; class Intermediate : public Base { public: void show() { cout << "Intermediate class show function" << endl; } }; class Derived : public Intermediate { public: void reveal() { cout << "Derived class reveal function" << endl; } }; int main() { Derived obj; obj.display(); obj.show(); obj.reveal(); return 0; }

22) Write a program to implement multiple inheritance

#include <iostream> using namespace std; class Base1 { public: void display() { cout << "Base1 class display function" << endl; } }; class Base2 { public: void show() { cout << "Base2 class show function" << endl; } }; class Derived : public Base1, public Base2 { public: void reveal() { cout << "Derived class reveal function" << endl; } }; int main() { Derived obj; obj.display(); obj.show(); obj.reveal(); return 0; }

23) Write a program to implement Hybrid inheritance

#include <iostream> using namespace std; class Base { public: void display() { cout << "Base class display function" << endl; } }; class Derived1 : public Base { public: void show() { cout << "Derived1 class show function" << endl; } }; class Derived2 : public Base { public: void reveal() { cout << "Derived2 class reveal function" << endl; } }; class Hybrid : public Derived1, public Derived2 { public: void expose() { cout << "Hybrid class expose function" << endl; } }; int main() { Hybrid obj; obj.Derived1::display(); // Ambiguity resolution obj.show(); obj.reveal(); obj.expose(); return 0; }

24) Write a program to implement the concept of virtual base class

25) Write a program to implement the concept in derived class
